<div id="text-container"></div>

<script>
  const textContainer = document.getElementById('text-container');
  const textToType = "hello, type this text!";
  let currentIndex = 0;

  // Initialize text container with spans
  textContainer.innerHTML = textToType.split('').map(char => `<span>${char}</span>`).join('');

  document.addEventListener('keydown', handleKeyPress);
  

  function handleKeyPress(event) {
    if (currentIndex < textToType.length) {
      const currentChar = textToType[currentIndex];
      const typedChar = event.key;

      if (typedChar === currentChar) {
        textContainer.children[currentIndex].classList.add('correct');
        currentIndex++;
      } else {
        textContainer.children[currentIndex].classList.add('incorrect');
        currentIndex++;
      }
    }
  }
</script>
